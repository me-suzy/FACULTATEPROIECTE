
	FUNCTION GET_IDS_FOR_GEN_ID( genID IN NUMBER, sSep IN VARCHAR2) RETURN VARCHAR2
AS
  L_STR  VARCHAR2(2000) DEFAULT NULL;
  L_SEP  VARCHAR2(10) DEFAULT NULL;
BEGIN
   L_SEP := '';
   FOR X IN ( SELECT ID FROM BC_HISTORY WHERE GENERATION_ID = genID) LOOP
       L_STR := L_STR || L_SEP || to_char( X.ID);
       L_SEP := sSep;
   END LOOP;
   RETURN L_STR;
END;
